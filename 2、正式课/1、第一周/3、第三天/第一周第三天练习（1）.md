```
    var i = 5;
    function fn(i) {
        return function (n) {
            console.log(n + (++i));
        }
    }
    var f = fn(3);
    f(4);
    fn(3)(6);
    fn(2)(6);
    f(2);
    console.log(i);
```

```
            console.log(n + (--i))
        }
    }
    var f = fn();
    f();
    f(3);
    fn()(2);
    fn()(3);
    f(4)
```
